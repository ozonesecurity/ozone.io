We really do mean simple:

What if connecting to RTSP cameras were as simple as:


  Application app;

  // Two RTSP sources 
  NetworkAVInput cam1( "cam1", "rtsp://170.93.143.139:1935/rtplive/0b01b57900060075004d823633235daa" );
  NetworkAVInput cam2("cam2","rtsp://170.93.143.139:1935/rtplive/e0ffa81e00a200ab0050fa36c4235c0a");
  app.addThread( &cam1 );
  app.addThread( &cam2 );

And doing Face detection on them was as simple as:


    // Detect faces for cam1
    FaceDetector faceDetector1( "modectcam1" );
    faceDetector1.registerProvider( cam1 );
    app.addThread( &faceDetector1 );

    // Detect any motion for cam2
    MotionDetector motionDetector2( "modectcam2" );
    motionDetector2.registerProvider( cam2 );
    app.addThread( &motionDetector2 );

And stitching arbirary streams together for a montage was as easy as:


// Let's make a mux/stitched handler for cam1 and cam2 and its debugs
    MatrixVideo matrixVideo( "quadcammux", PIX_FMT_YUV420P, 640, 480, FrameRate( 1, 10 ), 2, 2 );
    matrixVideo.registerProvider( cam1 );
    matrixVideo.registerProvider( *faceDetector1.deltaImageSlave() );
    matrixVideo.registerProvider( cam2 );
    matrixVideo.registerProvider( *motionDetector2.deltaImageSlave() );
    app.addThread( &matrixVideo );

And making all of this renderable on a browser without any plugins is as simple as:


    HttpController httpController( "watch", 9292 );
    httpController.addStream("watchcam1",cam1); // watch livestream of cam1
    httpController.addStream("watchcam2",cam2); // watch livestream of cam2
    httpController.addStream( "debug", matrixVideo ); // watch a 4x4 mux stiched from cam1,cam2 and their debug frames
    httpController.addStream( "debug", faceDetector1 );
    httpController.addStream( "debug", motionDetector2 );
    listener.addController( &httpController );
    app.run();
