330 {
331  Debug( 2, "Received RTMP request: (%zd bytes)", request.size() );
332  Hexdump( 3, request.data(), min(request.size(),128) );
333 
334  if ( mRequest.size() )
335  {
336  request = mRequest + request;
337  mRequest.clear();
338  }
339 
340  switch( mState )
341  {
342  case UNITIALIZED :
343  {
344  // Look for C0 & C1
345  if ( request[0] != CS0 )
346  {
347  Error( "Unexpected value received %02x", request[0] );
348  // Close connection
349  return( false );
350  }
351  Debug( 2, "Got RTMP C0 packet" );
352  request.consume( sizeof(CS0) );
353 
354  // Fallthru
355  }
356  case GOT_C0 :
357  {
358  if ( request.size() < sizeof(CS1) )
359  {
360  mRequest = request;
361  mState = GOT_C0;
362  return( true );
363  }
364 
365  Debug( 2, "Got RTMP C1 packet" );
366  memcpy( &mC1, request.data(), sizeof(mC1) );
367  request.consume( sizeof(mC1) );
368 
369  if ( mC1.zero & 0xff000000 )
370  {
371  // FP9 Handshake
372  Debug( 2, "Using FP9 handshake" );
373  mFP9 = true;
374  }
375 
376  sendResponse( &CS0, sizeof(CS0) );
377 
378  mS1.time = htobe32(rtmpTimestamp32());
379  if ( mFP9 )
380  mS1.zero = htobe32(0x03050101);
381  else
382  mS1.zero = 0;
383  for ( int i = 0; i < sizeof(mS1.random); i++ )
384  mS1.random[i] = 0;
385  //mS1.random[i] = rand();
386 
387  if ( mFP9 )
388  {
389  uint8_t *serverSig = (uint8_t *)&mS1;
390 
391  mServerDigestPosition = calculateDigestOffset( serverSig, OZ_RTMP_SIG_SIZE, 1 );
392  Debug( 3, "Server digest offset: %d", mServerDigestPosition );
393 
394  calculateDigest( mServerDigestPosition, serverSig, GenuineFMSKey, 36, serverSig+mServerDigestPosition );
395 
396  Debug( 3, "Initial server digest: " );
397  Hexdump( 3, serverSig + mServerDigestPosition, SHA256_DIGEST_LENGTH );
398  }
399 
400  Debug( 2, "Sending RTMP S1 packet" );
401  sendResponse( &mS1, sizeof(mS1) );
402 
403  if ( mFP9 )
404  {
405  uint8_t *clientSig = (uint8_t *)&mC1;
406  uint8_t digestResp[SHA256_DIGEST_LENGTH];
407  uint8_t *signatureResp = NULL;
408 
409  /* we have to use this signature now to find the correct algorithms for getting the digest and DH positions */
410  int clientDigestPosition = calculateDigestOffset( clientSig, OZ_RTMP_SIG_SIZE, 1 );
411 
412  if ( !verifyDigest( clientDigestPosition, clientSig, GenuineFPKey, 30 ) )
413  {
414  Debug( 2, "Trying alternate position for client digest" );
415 
416  clientDigestPosition = calculateDigestOffset( clientSig, OZ_RTMP_SIG_SIZE, 2 );
417

419  {
420  Error( "Unable to verify client digest" );
421  return false;
422  }
423  }
424 
425  /* calculate response now */
426  signatureResp = clientSig+OZ_RTMP_SIG_SIZE-SHA256_DIGEST_LENGTH;
427 
428  HMACsha256( clientSig+clientDigestPosition, SHA256_DIGEST_LENGTH, GenuineFMSKey, sizeof(GenuineFMSKey), digestResp );
429  HMACsha256( clientSig, OZ_RTMP_SIG_SIZE - SHA256_DIGEST_LENGTH, digestResp, SHA256_DIGEST_LENGTH, signatureResp );
430 
431  /* some info output */
432  Debug( 3, "Calculated digest key from secure key and server digest: " );
433  Hexdump( 3, digestResp, SHA256_DIGEST_LENGTH );
434 
435  Debug( 3, "Server signature calculated:" );
436  Hexdump( 3, signatureResp, SHA256_DIGEST_LENGTH );
437  }
438 
439  Debug( 2, "Sending RTMP S2 packet (reflecting C1 packet)" );
440  sendResponse( &mC1, sizeof(mC1) ); // Same as S2
441 
442  mState = SENT_S2;
443  return( true );
444  break;
445  }
446  case SENT_S2 :
447  {
448  if ( request.size() < sizeof(CS2) )
449  {
450  mRequest = request;
451  return( true );
452  }
453 
454  Debug( 2, "Got RTMP C2 packet" );
455  memcpy( &mC2, request.data(), sizeof(mC2) );
456  request.consume( sizeof(mC2) );
457 
458  if ( mFP9 )
459  {

461  uint8_t *clientSig = (uint8_t *)&mC2;
462  uint8_t signature[SHA256_DIGEST_LENGTH];
463  uint8_t digest[SHA256_DIGEST_LENGTH];
464 
465  Debug( 3, "Client sent signature:" );
466  Hexdump( 3, clientSig+(OZ_RTMP_SIG_SIZE-SHA256_DIGEST_LENGTH), SHA256_DIGEST_LENGTH);

471
