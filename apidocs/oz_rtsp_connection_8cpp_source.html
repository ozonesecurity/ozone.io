104 bool RtspConnection::handleRequest( const std::string &request )
105 {
106  Debug( 2, "Handling RTSP request: %s (%zd bytes)", request.c_str(), request.size() );
107 
108  StringTokenList lines( request, "\r\n" );
109  if ( lines.size() <= 0 )
110  {
111  Error( "Unable to split request '%s' into tokens", request.c_str() );
112  return( false );
113  }
114 
115  StringTokenList parts( lines[0], " " );
116  if ( parts.size() != 3 )
117  {
118  Error( "Unable to split request part '%s' into tokens", lines[0].c_str() );
119  return( false );
120  }
121 
122  std::string requestType = parts[0];
123  Debug( 4, "Got request '%s'", requestType.c_str() );
124  std::string requestUrl = parts[1];
125  Debug( 4, "Got requestUrl '%s'", requestUrl.c_str() );
126  std::string requestVer = parts[2];
127  Debug( 4, "Got requestVer '%s'", requestVer.c_str() );
128  if ( requestVer != "RTSP/1.0" )
129  {
130  Error( "Unexpected RTSP version '%s'", requestVer.c_str() );
131  return( false );
132  }
133 
134  // Extract headers from request
135  Headers requestHeaders;
136  for ( int i = 1; i < lines.size(); i++ )
137  {
138  StringTokenList parts( lines[i], ": " );
139  if ( parts.size() != 2 )
140  {
141  Error( "Unable to split request header '%s' into tokens", lines[i].c_str() );
142  return( false );
143  }
144  Debug( 4, "Got header '%s', value '%s'", parts[0].c_str(), parts[1].c_str() );
145  requestHeaders.insert( Headers::value_type( parts[0], parts[1] ) );
146  }
147 
148  if ( requestHeaders.find("CSeq") == requestHeaders.end() )
149  {
150  Error( "No CSeq header found" );
151  return( false );
152  }
153  Debug( 4, "Got sequence number %s", requestHeaders["CSeq"].c_str() );
154 
155  uint32_t session = 0;
156  if ( requestHeaders.find("Session") != requestHeaders.end() )
157  {
158  Debug( 4, "Got session header, '%s', passing to session", requestHeaders["Session"].c_str() );
159  session = strtol( requestHeaders["Session"].c_str(), NULL, 16 );
160  }
161 
162  Headers responseHeaders;
163  responseHeaders.insert( Headers::value_type( "CSeq", requestHeaders["CSeq"] ) );
164  if ( requestType == "OPTIONS" )
165  {
166  responseHeaders.insert( Headers::value_type( "Public", "DESCRIBE, SETUP, PLAY, GET_PARAMETER, TEARDOWN" ) );
167  return( sendResponse( responseHeaders ) );
168  }
169  else if ( requestType == "DESCRIBE" )
170  {
171  FeedProvider *provider = validateRequestUrl( requestUrl );
172 
173  if ( !provider )
174  {
175  sendResponse( responseHeaders, "", 404, "Not Found" );
176  return( false );
177  }
178 
179  const VideoProvider *videoProvider = dynamic_cast<const VideoProvider *>(provider);
180 
181  int codec = AV_CODEC_ID_MPEG4;
182  int width = videoProvider->width();
183  int height = videoProvider->height();
184  FrameRate frameRate = 15;
185  //FrameRate frameRate = videoProvider->frameRate();
186  int bitRate = 90000;
187  int quality = 70;
188 
189  std::string sdpFormatString =
190  "v=0\r\n"
191  "o=- %jd %jd IN IP4 %s\r\n"
192  "s=ZoneMinder Stream\r\n"
193  "i=Media Streamers\r\n"
194  "c=IN IP4 0.0.0.0\r\n"
195  "t=0 0\r\n"
196  "a=control:*\r\n"
197  "a=range:npt=0.000000-\r\n";
198  uint64_t now64 = time64();
199  char hostname[HOST_NAME_MAX] = "";
200  if ( gethostname( hostname, sizeof(hostname) ) < 0 )
201  Fatal( "Can't gethostname: %s", strerror(errno) );
202 
203  std::string sdpString = stringtf( sdpFormatString, now64, now64, hostname );
204 
205  if ( codec == AV_CODEC_ID_H264 )
206  {
207  if ( provider->cl4ss() == "RawH264Input" )
208  {
209  std::string encoderKey = H264Relay::getPoolKey( provider->identity(), width, height, frameRate, bitRate, quality );
210  if ( !(mEncoder = Encoder::getPooledEncoder( encoderKey )) )
211  {
212  H264Relay *h264Relay = NULL;
213  mEncoder = h264Relay = new H264Relay( provider->identity(), width, height, frameRate, bitRate, quality );
214  mEncoder->registerProvider( *provider );
215  Encoder::poolEncoder( mEncoder );
216  h264Relay->start();
217  }
218  sdpString += mEncoder->sdpString( 1 ); // XXX - Should be variable
219  responseHeaders.insert( Headers::value_type( "Content-length", stringtf( "%zd", sdpString.length() ) ) );
220  }
221  else
222  {
223  std::string encoderKey = H264Encoder::getPoolKey( provider->identity(), width, height, frameRate, bitRate, quality );
224  if ( !(mEncoder = Encoder::getPooledEncoder( encoderKey )) )
225  {
226  H264Encoder *h264Encoder = NULL;
227  mEncoder = h264Encoder = new H264Encoder( provider->identity(), width, height, frameRate, bitRate, quality );
228  mEncoder->registerProvider( *provider );

233  responseHeaders.insert( Headers::value_type( "Content-length", stringtf( "%zd", sdpString.length() ) ) );
234  }
235  }

239  if ( !(mEncoder = Encoder::getPooledEncoder( encoderKey )) )

243  mEncoder->registerProvider( *provider );
244  Encoder::poolEncoder( mEncoder );
245  mpegEncoder->start();
246  }
247  sdpString += mEncoder->sdpString( 1 ); // XXX - Should be variable
248  responseHeaders.insert( Headers::value_type( "Content-length", stringtf( "%zd", sdpString.length() ) ) );
249  }
250  return( sendResponse( responseHeaders, sdpString ) );
251  }
252  else if ( requestType == "SETUP" )
253  {
254  // These commands are handled by RTSP session so pass them on and send any required responses
255  RtspSession *rtspSession = 0;
256  if ( session )
257  {
258  rtspSession = mRtspController->getSession( session );
259  }
260  else
261  {
262  rtspSession = mRtspController->newSession( this, mEncoder );
263  }
264  if ( rtspSession->recvRequest( requestType, requestUrl, requestHeaders, responseHeaders ) )
265  return( sendResponse( responseHeaders ) );
266  return( false );
267  }
268  else if ( requestType == "PLAY" || requestType == "GET_PARAMETER" || requestType == "TEARDOWN" )
269  {
270  // These commands are handled by RTSP session so pass them on and send any required responses
271  RtspSession *rtspSession = 0;
272  if ( session )
273  {
274  rtspSession = mRtspController->getSession( session );
275  if ( rtspSession && rtspSession->recvRequest( requestType, requestUrl, requestHeaders, responseHeaders ) )
276  return( sendResponse( responseHeaders ) );
277  }
278  return( sendResponse( responseHeaders, "", 454, "Session Not Found" ) );
279  }
280  Error( "Unrecognised RTSP command '%s'", requestType.c_str() );
281  return( sendResponse( responseHeaders, "", 405, "Method not implemented" ) );

292 bool RtspConnection::recvRequest( ByteBuffer &buffer )
293 {
294  Debug( 2, "Received RTSP message, %zd bytes", buffer.size() );
295 
296  while( !buffer.empty() )
297  {
298  if ( buffer[0] == '$' || (!mRequest.empty() && mRequest[0] == '$') )
299  {
300  // Interleaved request
301  if ( mRequest )
302  {
303  buffer = mRequest+buffer;
304  mRequest.clear();
305  Debug( 6, "Merging with saved request, total bytes: %zd", buffer.size() );
306  }
307  Debug( 2, "Got RTP interleaved request" );
308  if ( buffer.size() <= 4 )
309  {
310  Debug( 6, "Request is incomplete, storing" );
311  mRequest = buffer;
312  return( true );
313  }
314  //int channel= buffer[1];
315  uint16_t rawLength;
316  memcpy( &rawLength, &buffer[2], sizeof(rawLength) );
317  uint16_t length = be16toh( rawLength );
318  if ( (length+2) > buffer.size() )
319  {
320  Debug( 6, "Request is incomplete, storing" );
321  Hexdump( 2, buffer.data(), buffer.size() );
322  mRequest = buffer;
323  return( true );
324  }
325  ByteBuffer request = buffer.range( 0, 4+length );
326  buffer.consume( 4+length );
327  if ( !handleRequest( request ) )
328  return( false );
329  }
330  else
331  {
332  Debug( 2, "Got RTSP request" );
333 
334  std::string request( reinterpret_cast<const char *>(buffer.data()), buffer.size() );
335 
336  if ( mRequest.size() > 0 )
337  {
338  //request = mRequest+request;
339  request = std::string( reinterpret_cast<const char *>(mRequest.data()), mRequest.size() )+request;
340  mRequest.clear();
341  Debug( 6, "Merging with saved request, total: %s", request.c_str() );
342  }
343 
344  const char *endOfMessage = strstr( request.c_str(), "\r\n\r\n" );
345 
346  if ( !endOfMessage )
347  {
348  Debug( 6, "Request '%s' is incomplete, storing", request.c_str() );
349  mRequest = ByteBuffer( reinterpret_cast<const unsigned char *>(request.data()), request.size() );
350  return( true );
351  }
352 
353  size_t messageLen = endOfMessage-request.c_str();
354  request.erase( messageLen );
355  buffer.consume( messageLen+4 );
356  if ( !handleRequest( request ) )
357  return( false );
358  }
359  }
360  return( true );
361 }
