82 bool RtpSession::updateSeq( uint16_t seq )
83 {
84  uint16_t uDelta = seq - mMaxSeq;
85 
86  // Source is not valid until MIN_SEQUENTIAL packets with
87  // sequential sequence numbers have been received.
88  Debug( 5, "Seq: %d", seq );
89 
90  if ( mProbation)
91  {
92  // packet is in sequence
93  if ( seq == mMaxSeq + 1)
94  {
95  Debug( 3, "Sequence in probation %d, in sequence", mProbation );
96  mProbation--;
97  mMaxSeq = seq;
98  if ( mProbation == 0 )
99  {
100  init( seq );
101  mReceivedPackets++;
102  return( true );
103  }
104  }
105  else
106  {
107  Warning( "Sequence in probation %d, out of sequence", mProbation );
108  mProbation = MIN_SEQUENTIAL - 1;
109  mMaxSeq = seq;
110  return( false );
111  }
112  return( true );
113  }
114  else if ( uDelta < MAX_DROPOUT )
115  {
116  if ( uDelta == 1 )
117  {
118  Debug( 3, "Packet in sequence, gap %d", uDelta );
119  }
120  else
121  {
122  Warning( "Packet in sequence, gap %d", uDelta );
123  }
124 
125  // in order, with permissible gap
126  if ( seq < mMaxSeq )
127  {
128  // Sequence number wrapped - count another 64K cycle.
129  mCycles += OZ_RTP_SEQ_MOD;
130  }
131  mMaxSeq = seq;
132  }
133  else if ( uDelta <= OZ_RTP_SEQ_MOD - MAX_MISORDER )
134  {
135  Warning( "Packet out of sequence, gap %d", uDelta );
136  // the sequence number made a very large jump
137  if ( seq == mBadSeq )
138  {
139  Debug( 3, "Restarting sequence" );
140  // Two sequential packets -- assume that the other side
141  // restarted without telling us so just re-sync
142  // (i.e., pretend this was the first packet).
143  init( seq );
144  }
145  else
146  {
147  mBadSeq = (seq + 1) & (OZ_RTP_SEQ_MOD-1);
148  return( false );
149  }
150  }
151  else
152  {
153  Warning( "Packet duplicate or reordered, gap %d", uDelta );
154  // duplicate or reordered packet
155  return( false );
156  }
157  mReceivedPackets++;
158  return( uDelta==1?true:false );
159 }

166 void RtpSession::updateJitter( const RtpDataHeader *header )
167 {
168  if ( mRtpFactor > 0 )
169  {
170  Debug( 5, "Delta rtp = %.6f", tvDiffSec( mBaseTimeReal ) );
171  uint32_t localTimeRtp = mBaseTimeRtp + uint32_t( tvDiffSec( mBaseTimeReal ) * mRtpFactor );
172  Debug( 5, "Local RTP time = %x", localTimeRtp );
173  Debug( 5, "Packet RTP time = %x", ntohl(header->timestampN) );
174  uint32_t packetTransit = localTimeRtp - ntohl(header->timestampN);
175  Debug( 5, "Packet transit RTP time = %x", packetTransit );
176 
177  if ( mTransit > 0 )
178  {
179  // Jitter
180  int d = packetTransit - mTransit;
181  Debug( 5, "Jitter D = %d", d );
182  if ( d < 0 )
183  d = -d;
184  //mJitter += (1./16.) * ((double)d - mJitter);
185  mJitter += d - ((mJitter + 8) >> 4);
186  }
187  mTransit = packetTransit;
188  }
189  else
190  {
191  mJitter = 0;
192  }
193  Debug( 5, "RTP Jitter: %d", mJitter );
