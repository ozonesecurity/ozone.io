92 /*
93  * Empty the output buffer --- called whenever buffer fills up.
94  *
95  * In typical applications, this should write the entire output buffer
96  * (ignoring the current state of next_output_byte & free_in_buffer),
97  * reset the pointer & count to the start of the buffer, and return TRUE
98  * indicating that the buffer has been dumped.
99  *
100  * In applications that need to be able to suspend compression due to output
101  * overrun, a FALSE return indicates that the buffer cannot be emptied now.
102  * In this situation, the compressor will return to its caller (possibly with
103  * an indication that it has not accepted all the supplied scanlines). The
104  * application should resume compression after it has made more room in the
105  * output buffer. Note that there are substantial restrictions on the use of
106  * suspension --- see the documentation.
107  *
108  * When suspending, the compressor will back up to a convenient restart point
109  * (typically the start of the current MCU). next_output_byte & free_in_buffer
110  * indicate where the restart point will be if the current call returns FALSE.
111  * Data beyond this point will be regenerated after resumption, so do not
112  * write it out when emptying the buffer externally.
113  */
114

216 /*
217  * Fill the input buffer --- called whenever buffer is emptied.
218  *
219  * In typical applications, this should read fresh data into the buffer
220  * (ignoring the current state of next_input_byte & bytes_in_buffer),
221  * reset the pointer & count to the start of the buffer, and return TRUE
222  * indicating that the buffer has been reloaded. It is not necessary to
223  * fill the buffer entirely, only to obtain at least one more byte.
224  *
225  * There is no such thing as an EOF return. If the end of the file has been
226  * reached, the routine has a choice of ERREXIT() or inserting fake data into
227  * the buffer. In most cases, generating a warning message and inserting a
228  * fake EOI marker is the best course of action --- this will allow the
229  * decompressor to output however much of the image is there. However,
230  * the resulting error message is misleading if the real problem is an empty
231  * input file, so we handle that case specially.
232  *
233  * In applications that need to be able to suspend compression due to input
234  * not being available yet, a FALSE return indicates that no more data can be
235  * obtained right now, but more may be forthcoming later. In this situation,
236  * the decompressor will return to its caller (with an indication of the
237  * number of scanlines it has read, if any). The application should resume
238  * decompression after it has loaded more data into the input buffer. Note
239  * that there are substantial restrictions on the use of suspension --- see
240  * the documentation.
241  *
242  * When suspending, the decompressor will back up to a convenient restart point
243  * (typically the start of the current MCU). next_input_byte & bytes_in_buffer
244  * indicate where the restart point will be if the current call returns FALSE.
245  * Data beyond this point must be rescanned after resumption, so move it to
246  * the front of the buffer rather than discarding it.
247  */
248 
249 static boolean fill_input_buffer (j_decompress_ptr cinfo)
250 {
251  mem_src_ptr src = (mem_src_ptr) cinfo->src;
252  size_t nbytes;
253 
254  memcpy( src->buffer, src->inbuffer, (size_t) src->inbuffer_size );
255  nbytes = src->inbuffer_size;
256 
257  if ( nbytes <= 0 )
258  {
259  if ( src->start_of_data ) /* Treat empty input file as fatal error */
260  ERREXIT(cinfo, JERR_INPUT_EMPTY);
261  WARNMS(cinfo, JWRN_JPEG_EOF);
262  /* Insert a fake EOI marker */
263  src->buffer[0] = (JOCTET) 0xFF;
264  src->buffer[1] = (JOCTET) JPEG_EOI;
265  nbytes = 2;
266  }
267 
268  src->pub.next_input_byte = src->buffer;
269  src->pub.bytes_in_buffer = nbytes;
270  src->start_of_data = FALSE;
271 
272  return( TRUE );
273 }

288 static void skip_input_data (j_decompress_ptr cinfo, long num_bytes)
289 {
290  mem_src_ptr src = (mem_src_ptr) cinfo->src;
291 
292  /* Just a dumb implementation for now. Could use fseek() except
293  * it doesn't work on pipes. Not clear that being smart is worth
294  * any trouble anyway --- large skips are infrequent.
295  */
296  if ( num_bytes > 0 )
297  {
298  while ( num_bytes > (long) src->pub.bytes_in_buffer )
299  {
300  num_bytes -= (long) src->pub.bytes_in_buffer;
301  (void) fill_input_buffer(cinfo);
302  /* note we assume that fill_input_buffer will never return FALSE,
303  * so suspension need not be handled.
304  */
305  }
306  src->pub.next_input_byte += (size_t) num_bytes;
307  src->pub.bytes_in_buffer -= (size_t) num_bytes;
308  }
309 }
