46 bool RtspSession::recvRequest( const std::string &requestType, const std::string &requestUrl, const Connection::Headers &requestHeaders, Connection::Headers &responseHeaders )
47 {
48  Debug( 2, "Session %d - Processing RTSP request: %s", mSession, requestType.c_str() );
49 
50  if ( requestType == "SETUP" )
51  {
52  std::string requestHost, requestStreamName, requestStreamSource;
53  int trackId = -1;
54  FeedProvider *provider = mConnection->validateRequestUrl( requestUrl, &requestHost, &requestStreamName, &requestStreamSource, &trackId );
55  if ( !provider )
56  {
57  mConnection->sendResponse( responseHeaders, "", 404, "Not Found" );
58  return( false );
59  }
60  if ( !provider->ready() )
61  {
62  mConnection->sendResponse( responseHeaders, "", 503, "Stream not ready" );
63  return( false );
64  }
65 
66  if ( trackId < 0 || trackId > 1 )
67  {
68  Error( "Got invalid trackID %d from request URL '%s'", trackId, requestUrl.c_str() );
69  return( false );
70  }
71 
72  if ( mRtspStreams.find( trackId ) != mRtspStreams.end() )
73  {
74  Error( "Got RTSP SETUP for existing trackID %d", trackId );
75  return( false );
76  }
77 
78  Connection::Headers::const_iterator iter = requestHeaders.find("Transport");
79  if ( iter == requestHeaders.end() )
80  {
81  Error( "No Transport header found in SETUP" );
82  return( false );
83  }
84  std::string transportSpec = iter->second;
85 
86  RtspStream *stream = newStream( trackId, mEncoder, transportSpec );
87 
88  transportSpec.clear();
89  transportSpec += "RTP/AVP";
90  if ( stream->lowerTransport() == RtspStream::LOWTRANS_TCP )
91  {
92  transportSpec += "/TCP";
93  }
94  else
95  {
96  transportSpec += "/UDP";
97  if ( stream->delivery() == RtspStream::DEL_UNICAST )
98  transportSpec += ";unicast";
99  else
100  transportSpec += ";multicast";
101  }
102  if ( stream->interleaved() )
103  {
104  transportSpec += stringtf( ";interleaved=%d-%d", stream->channel(0), stream->channel(1) );
105  }
106  else
107  {
108  transportSpec += stringtf( ";client_port=%d-%d", stream->destPort(0), stream->destPort(1) );
109  transportSpec += stringtf( ";server_port=%d-%d", stream->sourcePort(0), stream->sourcePort(1) );
110  }
111  transportSpec += stringtf( ";ssrc=%08X", stream->ssrc() );
112  transportSpec += ";mode=\"PLAY\"";
113  responseHeaders.insert( Connection::Headers::value_type( "Session", stringtf( "%08X", mSession ) ) );
114  responseHeaders.insert( Connection::Headers::value_type( "Transport", transportSpec ) );
115 
116  state( READY );
117  return( true );
118  }
119  else if ( requestType == "PLAY" )
120  {
121  std::string rtpInfo;
122  for ( RtspStreams::iterator iter = mRtspStreams.begin(); iter != mRtspStreams.end(); iter++ )
123  {
124  const RtspStream *stream = iter->second;
125  if ( rtpInfo.size() )
126  rtpInfo += ",";
127  rtpInfo += stringtf( "url=%s/trackID=%d;seq=0;rtptime=0", requestUrl.c_str(), stream->trackId() );
128  }
129  responseHeaders.insert( Connection::Headers::value_type( "Session", stringtf( "%08X", mSession ) ) );
130  responseHeaders.insert( Connection::Headers::value_type( "Range", "npt=0-" ) );
131  responseHeaders.insert( Connection::Headers::value_type( "RTP-Info", rtpInfo ) );
132  for ( RtspStreams::iterator iter = mRtspStreams.begin(); iter != mRtspStreams.end(); iter++ )
133  iter->second->start();
134  state( PLAYING );
135  return( true );
136  }
137  else if ( requestType == "GET_PARAMETER" )
138  {
139  FeedProvider *provider = mConnection->validateRequestUrl( requestUrl );
140  if ( provider )
141  {
142  responseHeaders.insert( Connection::Headers::value_type( "Session", stringtf( "%08X", mSession ) ) );
143  return( true );
144  }
145  mConnection->sendResponse( responseHeaders, "", 451, "Parameter Not Understood" );
146  return( false );
147  }
148  else if ( requestType == "TEARDOWN" )
149  {
150  Debug( 1, "Stopping streams" );
151  for ( RtspStreams::iterator iter = mRtspStreams.begin(); iter != mRtspStreams.end(); iter++ )
152  iter->second->stop();
153  Debug( 1, "Waiting for streams" );
154  for ( RtspStreams::iterator iter = mRtspStreams.begin(); iter != mRtspStreams.end(); iter++ )
155  iter->second->join();
156  for ( RtspStreams::iterator iter = mRtspStreams.begin(); iter != mRtspStreams.end(); iter++ )
157  delete iter->second;
158  mRtspStreams.clear();
159  state( INIT );
160  return( true );
161  }
162  Error( "Unrecognised RTSP command '%s'", requestType.c_str() );
163  return( false );
164 }

176 RtspStream *RtspSession::newStream( int trackId, Encoder *encoder, const std::string &transportSpec )
177 {
178  StringTokenList transportParts( transportSpec, ";" );
179  if ( transportParts.size() < 1 )
180  {
181  Error( "Unable to parse transport data from transport '%s'", transportSpec.c_str() );
182  return( NULL );
183  }
184 
185  const std::string &protocolSpec = transportParts[0];
186  StringTokenList protocolParts( protocolSpec, "/" );
187  if ( protocolParts.size() < 2 )
188  {
189  Error( "Unable to parse transport data from protocol '%s'", protocolSpec.c_str() );
190  return( NULL );
191  }
192 
193  const std::string transport = protocolParts[0];
194  const std::string profile = protocolParts[1];
195  const std::string lowerTransport = protocolParts.size()>2?protocolParts[2]:"UDP";
196 
197  Debug( 3, "Got transport '%s'", transport.c_str() );
198  Debug( 3, "Got profile '%s'", profile.c_str() );
199  Debug( 3, "Got lowerTransport '%s'", lowerTransport.c_str() );
200 
201  StringTokenList::TokenList tokens = transportParts.tokens();
202  tokens.pop_front();
203 
204  RtspStream *rtspStream = new RtspStream( this, trackId, mConnection, encoder, transport, profile, lowerTransport, tokens );
205  mRtspStreams.insert( RtspStreams::value_type( trackId, rtspStream ) );
206 
207  return( rtspStream );
208 }
