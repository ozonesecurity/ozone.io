68  int mapFd = open( memFile, O_RDWR, (mode_t)0600 );
69  if ( mapFd < 0 )
70  {
71  Debug( 3, "Can't open memory map file %s: %s", memFile, strerror(errno) );
72  return( false );
73  }
74  struct stat mapStat;
75  if ( fstat( mapFd, &mapStat ) < 0 )
76  Fatal( "Can't stat memory map file %s: %s", memFile, strerror(errno) );
77  if ( mapStat.st_size == 0 )
78  {
79  close( mapFd );
80  //unlink( memFile );
81  return( false );
82  }
83  else if ( mapStat.st_size < memSize ) // Is allowed to be larger as not querying whole thing
84  {
85  Error( "Got unexpected memory map file size %ld, expected %lu", mapStat.st_size, memSize );
86  }
87 
88  memPtr = (unsigned char *)mmap( NULL, memSize, PROT_READ|PROT_WRITE, MAP_SHARED|MAP_LOCKED, mapFd, 0 );
89  if ( memPtr == MAP_FAILED )
90  if ( errno == EAGAIN )
91  {
92  Debug( 1, "Unable to map file %s (%lu bytes) to locked memory, trying unlocked", memFile, memSize );
93  memPtr = (unsigned char *)mmap( NULL, memSize, PROT_READ|PROT_WRITE, MAP_SHARED, mapFd, 0 );
94  }
95  if ( memPtr == MAP_FAILED )
96  {
97  Debug( 3, "Can't map file %s (%lu bytes) to memory: %s(%d)", memFile, memSize, strerror(errno), errno );
98  if ( munmap( memPtr, memSize ) < 0 )
99  Fatal( "Can't munmap: %s", strerror(errno) );
100  close( mapFd );
101  //unlink( memFile );
102  return( false );
103  }
104 #else // OZ_MEM_MAPPED
105  int shmId = shmget( (config.shm_key&0xffff0000)|mMemoryKey, memSize, IPC_CREAT|0700 );
106  if ( shmId < 0 )
107  {
108  Debug( 3, "Can't shmget, probably not enough shared memory space free: %s", strerror(errno));
109  return( false );
110  }
111  memPtr = (unsigned char *)shmat( shmId, 0, 0 );
112  if ( memPtr < 0 )

115  return( false );
116  }
117 #endif // OZ_MEM_MAPPED
118 
119  memcpy( sharedData, (SharedData *)memPtr, sizeof(*sharedData) );
