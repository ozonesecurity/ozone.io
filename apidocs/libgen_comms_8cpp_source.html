958 int Select::wait()
959 {
960  struct timeval tempTimeout = mTimeout;
961  struct timeval *selectTimeout = mHasTimeout?&tempTimeout:NULL;
962 
963  fd_set rfds;
964  fd_set wfds;
965 
966  mReadable.clear();
967  FD_ZERO(&rfds);
968  for ( CommsSet::iterator iter = mReaders.begin(); iter != mReaders.end(); iter++ )
969  FD_SET((*iter)->getReadDesc(),&rfds);
970 
971  mWriteable.clear();
972  FD_ZERO(&wfds);
973  for ( CommsSet::iterator iter = mWriters.begin(); iter != mWriters.end(); iter++ )
974  FD_SET((*iter)->getWriteDesc(),&wfds);
975 
976  int nFound = select( mMaxFd+1, &rfds, &wfds, NULL, selectTimeout );
977  if( nFound == 0 )
978  {
979  Debug( 1, "Select timed out" );
980  }
981  else if ( nFound < 0)
982  {
983  Error( "Select error: %s", strerror(errno) );
984  }
985  else
986  {
987  for ( CommsSet::iterator iter = mReaders.begin(); iter != mReaders.end(); iter++ )
988  if ( FD_ISSET((*iter)->getReadDesc(),&rfds) )
989  mReadable.push_back( *iter );
990  for ( CommsSet::iterator iter = mWriters.begin(); iter != mWriters.end(); iter++ )
991  if ( FD_ISSET((*iter)->getWriteDesc(),&wfds) )
992  mWriteable.push_back( *iter );
993  }
