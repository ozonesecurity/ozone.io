64 int QueuedVideoFilter::run()
65 {
66  if ( waitForProviders() )
67  {
68  while ( !mStop )
69  {
70  mQueueMutex.lock();
71  if ( !mFrameQueue.empty() )
72  {
73  Debug( 3, "Got %zd frames on queue", mFrameQueue.size() );
74  for ( FrameQueue::iterator iter = mFrameQueue.begin(); iter != mFrameQueue.end(); iter++ )
75  {
76  distributeFrame( *iter );
77  mFrameCount++;
78  }
79  mFrameQueue.clear();
80  }
81  mQueueMutex.unlock();
82  checkProviders();
83  // Quite short so we can always keep up with the required packet rate for 25/30 fps
84  usleep( INTERFRAME_TIMEOUT );
85  }
86  }
87  FeedProvider::cleanup();
88  FeedConsumer::cleanup();
89  return( !ended() );
