68  Debug( 3, "Got %zd frames on queue", mFrameQueue.size() );

70  {
71  // Only operate on video frames
72  const VideoFrame *frame = dynamic_cast<const VideoFrame *>(iter->get());
73  if ( frame )
74  {
75  Image image( pixelFormat, width, height, frame->buffer().data() );
76  Image convImage( Image::FMT_RGB, image );
77  dlib::array2d<dlib::rgb_pixel> img( height, width );
78  memcpy( image_data( img ), convImage.buffer().data(), convImage.buffer().size() );

82  Info( "%jd @ %ju: Got %jd faces", frame->id(), frame->timestamp(), dets.size() );
83 
84  if ( dets.size() > 0 && mFaceMarkup != OZ_FACE_MARKUP_NONE )
85  {
86  // Now we will go ask the shape_predictor to tell us the pose of
87  // each face we detected.
88  std::vector<dlib::full_object_detection> shapes;
89  const dlib::rgb_pixel color = dlib::rgb_pixel(0,255,0);
90  typedef std::pair<dlib::point,dlib::point> line_t;
91  std::vector<line_t> lines;
92  for ( unsigned int i = 0; i < dets.size(); i++ )
93  {
94  if ( mFaceMarkup & OZ_FACE_MARKUP_OUTLINE )

119  // Line from the nose to the bottom part above
120  lines.push_back(line_t(shape.part(30), shape.part(35)));
121

151  dlib::draw_line( img, line.first, line.second, color );

157  }
