86 void Mp4FileOutput::closeFile( AVFormatContext *outputContext )

88  /* write the trailer, if any. the trailer must be written

93 
94  for ( int i = 0; i < outputContext->nb_streams; i++ )
95  {
96  /* close each codec */

102

107  av_free( outputContext );
