326 uint32_t MotionDetector::detectMotion( const Image &compImage, ZoneSet &motionZones )
327 {
328  bool alarm = false;
329  uint64_t score = 0;
330 
331  if ( mZones.size() <= 0 ) return( alarm );
332 
333  if ( config.record_diag_images )
334  {
335  mCompImageSlave->relayImage( compImage );
336  //compImage.writeJpeg( stringtf( "%s/diag-c.jpg", config.dir_events ) );
337  if ( writeImages )
338  compImage.writeJpeg( stringtf( "%s/comp-%s-%ju.jpg", writeLocation.c_str(), mName.c_str(), mFrameCount ), 100 );
339  mRefImageSlave->relayImage( mRefImage );
340  //mRefImage.writeJpeg( stringtf( "%s/diag-r.jpg", config.dir_events ) );
341  if ( writeImages )
342  mRefImage.writeJpeg( stringtf( "%s/ref2-%s-%ju.jpg", writeLocation.c_str(), mName.c_str(), mFrameCount ), 100 );
343  }
344 
345  // Get the difference between the captured image and the reference image
346  mDeltaCorrection = false;
347  Image *deltaImage = 0;
348  if ( mColourDeltas )
349  deltaImage = mRefImage.delta2( compImage, mDeltaCorrection );
350  else
351  deltaImage = mRefImage.yDelta2( compImage, mDeltaCorrection );
352  //Info( "DI: %dx%d - %d", deltaImage->width(), deltaImage->height(), deltaImage->format() );
353 
354  // Reduce the size of the diff image, also helps to despeckle
355  if ( mAnalysisScale > 1 )
356  deltaImage->shrink( mAnalysisScale );
357  if ( config.record_diag_images )
358  {
359  mDeltaImageSlave->relayImage( *deltaImage );
360  //deltaImage->writeJpeg( stringtf( "%s/diag-d.jpg", config.dir_events ), 100 );
361  if ( writeImages )
362  deltaImage->writeJpeg( stringtf( "%s/delta-%s-%ju.jpg", writeLocation.c_str(), mName.c_str(), mFrameCount ), 100 );
363  }
364  // Pre-populate the variance buffer with noise
365  if ( mVarBuffer.empty() )
366  mVarBuffer.fill( mNoiseLevelSq, deltaImage->pixels() );
367 
368  // Blank out all exclusion zones
369  for ( ZoneSet::iterator zoneIter = mZones.begin(); zoneIter!= mZones.end(); zoneIter++ )
370  {
371  Zone *zone = *zoneIter;
372  zone->clearAlarmed();
373  if ( !zone->isInactive() )
374  continue;
375  Debug( 3, "Blanking inactive zone %s", zone->label().c_str() );
376  deltaImage->fill( RGB_BLACK, zone->getPolygon() );
377  }
378 
379  // Check preclusive zones first
380  for ( ZoneSet::iterator zoneIter = mZones.begin(); zoneIter!= mZones.end(); zoneIter++ )
381  {
382  Zone *zone = *zoneIter;
383  if ( !zone->isPreclusive() )
384  continue;
385  Debug( 3, "Checking preclusive zone %s", zone->label().c_str() );
386  if ( zone->checkMotion( deltaImage, mVarBuffer ) )
387  {
388  Debug( 3, "Preclusive zone %s is alarmed, zone score = %d", zone->label().c_str(), zone->score() );
389  return( false );
390  }
391  }
392 
393  double topScore = -1.0;
394  Coord alarmCentre;
395 
396  // Find all alarm pixels in active zones
397  for ( ZoneSet::iterator zoneIter = mZones.begin(); zoneIter!= mZones.end(); zoneIter++ )
398  {
399  Zone *zone = *zoneIter;
400  if ( !zone->isActive() )
401  continue;
402  Debug( 3, "Checking active zone %s", zone->label().c_str() );
403  if ( zone->checkMotion( deltaImage, mVarBuffer ) )
404  {
405  alarm = true;
406  score += zone->score();
407  zone->setAlarmed();
408  Debug( 3, "Active zone %s is alarmed, zone score = %d", zone->label().c_str(), zone->score() );
409  motionZones.insert( zone );
410  }
411  }
412 
413  if ( alarm )
414  {
415  for ( ZoneSet::iterator zoneIter = mZones.begin(); zoneIter!= mZones.end(); zoneIter++ )
416  {
417  Zone *zone = *zoneIter;
418  if ( !zone->isInclusive() )
419  continue;
420  Debug( 3, "Checking inclusive zone %s", zone->label().c_str() );
421  if ( zone->checkMotion( deltaImage, mVarBuffer ) )
422  {
423  alarm = true;
424  score += zone->score();
425  zone->setAlarmed();
426  Debug( 3, "Inclusive zone %s is alarmed, zone score = %d", zone->label().c_str(), zone->score() );
427  motionZones.insert( zone );
428  }
429  }
430  }
431  else
432  {
433  // Find all alarm pixels in exclusive zones
434  for ( ZoneSet::iterator zoneIter = mZones.begin(); zoneIter!= mZones.end(); zoneIter++ )
435  {
436  Zone *zone = *zoneIter;
437  if ( !zone->isExclusive() )
438  continue;
439  Debug( 3, "Checking exclusive zone %s", zone->label().c_str() );
440  if ( zone->checkMotion( deltaImage, mVarBuffer ) )
441  {
442  alarm = true;
443  score += zone->score();
444  zone->setAlarmed();
445  Debug( 3, "Exclusive zone %s is alarmed, zone score = %d", zone->label().c_str(), zone->score() );
446  motionZones.insert( zone );
447  }
448  }
449  }
450 
451  if ( topScore > 0 )
452  {
453  Info( "Got alarm centre at %d,%d, at count %ju", alarmCentre.x(), alarmCentre.y(), mFrameCount );
454  }
455 
456  // Update the variance buffer based on the delta image
457  uint8_t *pDelta = deltaImage->buffer().head();
458  uint32_t *pBuf = mVarBuffer.head();
459  int deltaStep = deltaImage->pixelStep();
460  int varBufShift = getShift( mVarBlend, mFastStart?mFrameCount+1:0 );
461  int varDeltaShift = 16 - varBufShift;
462 
463  if ( deltaImage->format() == Image::FMT_GREY16 )
464  {
465  while( pDelta != deltaImage->buffer().tail() )
466  {
467  uint16_t delta = (((uint16_t)*pDelta)<<8)+*(pDelta+1);
468  //uint32_t long deltaSq = delta*delta*2;
469  uint64_t deltaSq = delta*delta;
470  *pBuf = *pBuf - (*pBuf >> varBufShift) + (deltaSq >> varDeltaShift);
471  // Add a noise threshold
472  if ( *pBuf < mNoiseLevelSq )
473  *pBuf = mNoiseLevelSq;
474  pDelta += deltaStep;
475  pBuf++;
476  }
477  }
478  else
479  {
480  while( pDelta != deltaImage->buffer().tail() )
481  {
482  uint8_t delta = *pDelta;
483  //uint16_t deltaSq = delta*delta*2;
484  uint16_t deltaSq = delta*delta;
485  *pBuf = *pBuf - (*pBuf >> varBufShift) + (deltaSq << varDeltaShift);
486  // Add a noise threshold
487  if ( *pBuf < mNoiseLevelSq )
488  *pBuf = mNoiseLevelSq;
489  pDelta += deltaStep;
490  pBuf++;
491  }
492  }
493 
494  if ( config.record_diag_images )
495  {
496  if ( !mVarImage )
497  mVarImage = new Image( Image::FMT_GREY, deltaImage->width(), deltaImage->height() );
498  else
499  mVarImage->erase();
500 
501  uint8_t *pVar = mVarImage->buffer().head();
502  pBuf = mVarBuffer.head();
503  int varStep = mVarImage->pixelStep();
504  while( pBuf != mVarBuffer.tail() )
505  {
506  *pVar = (uint8_t)sqrt(*pBuf>>16);
507  pVar += varStep;
508  pBuf++;
509  }
510  mVarImageSlave->relayImage( *mVarImage );
511  //mVarImage->writeJpeg( stringtf( "%s/diag-v.jpg", config.dir_events ), 100 );
512  if ( writeImages )
513  mVarImage->writeJpeg( stringtf( "%s/var-%s-%ju.jpg", writeLocation.c_str(), mName.c_str(), mFrameCount ), 100 );
514  }
515  delete deltaImage;
516  //mRefImage.blend( compImage, mRefBlend );
517 
518  // This is a small and innocent hack to prevent scores of 0 being returned in alarm state
519  return( score?score:alarm );
520 }
