39 {
40  Select select( 60 );
41  select.addWriter( mConnection->socket() );
42 
43  if ( !waitForProviders() )
44  return( 1 );
45 
46  while ( !mStop && select.wait() >= 0 )
47  {
48  if ( mStop )
49  break;
50  Select::CommsList writeable = select.getWriteable();
51  if ( writeable.size() <= 0 )
52  {
53  Error( "Writer timed out" );
54  mStop = true;
55  break;
56  }
57  mQueueMutex.lock();
58  if ( !mFrameQueue.empty() )
59  {
60  for ( FrameQueue::iterator iter = mFrameQueue.begin(); iter != mFrameQueue.end(); iter++ )
61  {
62  sendFrame( writeable, *iter );

64  }
65  mFrameQueue.clear();
66  }
67  mQueueMutex.unlock();
68  usleep( INTERFRAME_TIMEOUT );
69  }

117 bool HttpImageStream::sendFrame( Select::CommsList &writeable, FramePtr frame )
118 {
119  const ByteBuffer &packet = frame->buffer();
120 
121  std::string txHeaders = "--imgboundary\r\n";
122  txHeaders += "Content-Type: image/jpeg\r\n";
123  txHeaders += stringtf( "Content-Length: %zd\r\n\r\n", packet.size() );
124 
125  ByteBuffer txBuffer;
126  txBuffer.reserve( packet.size()+128 );
127  txBuffer.append( txHeaders.c_str(), txHeaders.size() );
128  txBuffer.append( packet.data(), packet.size() );
129  txBuffer.append( "\r\n", 2 );
130 
131  for ( Select::CommsList::iterator iter = writeable.begin(); iter != writeable.end(); iter++ )
132  {
133  if ( TcpInetSocket *socket = dynamic_cast<TcpInetSocket *>(*iter) )
134  {
135  if ( socket == mConnection->socket() )
136  {
137  int nBytes = socket->write( txBuffer.data(), txBuffer.size() );
138  const FeedFrame *sourceFrame = frame->sourceFrame();
139  Debug( 4, "Wrote %d bytes on sd %d, frame %ju<-%ju", nBytes, socket->getWriteDesc(), frame->id(), sourceFrame->id() );
140  if ( nBytes != txBuffer.size() )
141  {
142  Error( "Incomplete write, %d bytes instead of %zd", nBytes, packet.size() );
143  mStop = true;
144  return( false );
145  }
146  }
147  }
148  }
149  return( true );
150 }
