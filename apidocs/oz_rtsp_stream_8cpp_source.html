183 void RtspStream::packetiseFrame( FramePtr frame )
184 {
185  Debug( 1, "Got %zd byte frame to packetise", frame->buffer().size() );
186 
187  if ( mEncoder->cl4ss() == "H264Encoder" || mEncoder->cl4ss() == "H264Relay" )
188  {
189  const unsigned char *startPos = h264StartCode( frame->buffer().head(), frame->buffer().tail() );
190  while ( startPos < frame->buffer().tail() )
191  {
192  while( !*(startPos++) )
193  ;
194  const unsigned char *nextStartPos = h264StartCode( startPos, frame->buffer().tail() );
195 
196  //nal_send(s1, r, r1 - r, (r1 == buf1 + size));
197  //static void nal_send(AVFormatContext *s1, const uint8_t *buf, int size, int last)
198 
199  int frameSize = nextStartPos-startPos;
200 
201  if ( frameSize <= OZ_RTP_MAX_PACKET_SIZE )
202  {
203  unsigned char type = startPos[0] & 0x1F;
204  unsigned char nri = startPos[0] & 0x60;
205 
206  Debug( 2, "Packet type %d, nri %0x, size %d", type, nri, frameSize );
207 
208  //ff_rtp_send_data(s1, buf, size, last);
209  mRtpData->buildPacket( startPos, frameSize, frame->timestamp(), (nextStartPos == frame->buffer().tail()) );
210  }
211  else
212  {
213  unsigned char tempBuffer[OZ_RTP_MAX_PACKET_SIZE];
214 
215  unsigned char type = startPos[0] & 0x1F;
216  unsigned char nri = startPos[0] & 0x60;
217 
218  Debug( 2, "Packet type %d, nri %0x, size %d", type, nri, frameSize );
219 
220  tempBuffer[0] = 28; /* FU Indicator; Type = 28 ---> FU-A */
221  tempBuffer[0] |= nri;
222  tempBuffer[1] = type;
223  tempBuffer[1] |= 1 << 7;
224 
225  startPos += 1;
226  frameSize -= 1;
227  while ( (frameSize + 2) > OZ_RTP_MAX_PACKET_SIZE )
228  {
229  memcpy( &tempBuffer[2], startPos, OZ_RTP_MAX_PACKET_SIZE - 2);
230  mRtpData->buildPacket( tempBuffer, OZ_RTP_MAX_PACKET_SIZE, frame->timestamp(), false );
231  startPos += OZ_RTP_MAX_PACKET_SIZE - 2;
232  frameSize -= OZ_RTP_MAX_PACKET_SIZE - 2;
233  tempBuffer[1] &= ~(1 << 7);
234  }
235  tempBuffer[1] |= 1 << 6;
236  memcpy( &tempBuffer[2], startPos, frameSize );
237  mRtpData->buildPacket( tempBuffer, frameSize+2, frame->timestamp(), (nextStartPos == frame->buffer().tail()) );
238  }
239  startPos = nextStartPos;
240  }
241  }
242  else
243  {
244  if ( frame->buffer().size() <= OZ_RTP_MAX_PACKET_SIZE )
245  {
246  //ff_rtp_send_data(s1, buf, size, last);
247  mRtpData->buildPacket( frame->buffer().data(), frame->buffer().size(), frame->timestamp(), true );
248  }
249  else
250  {
251  unsigned char tempBuffer[OZ_RTP_MAX_PACKET_SIZE];
252  uint8_t *frameDataPtr = frame->buffer().data();
253  uint32_t frameSize = frame->buffer().size();
254  while ( frameSize > OZ_RTP_MAX_PACKET_SIZE )
255  {
256  memcpy( tempBuffer, frameDataPtr, OZ_RTP_MAX_PACKET_SIZE );
257  mRtpData->buildPacket( tempBuffer, OZ_RTP_MAX_PACKET_SIZE, frame->timestamp(), false );
258  frameDataPtr += OZ_RTP_MAX_PACKET_SIZE;
259  frameSize -= OZ_RTP_MAX_PACKET_SIZE;
260  }
261  memcpy( tempBuffer, frameDataPtr, frameSize );
262  mRtpData->buildPacket( tempBuffer, frameSize, frame->timestamp(), true );
263  }
264  }

302  Fatal( "Failed to connect RTP data socket" );

304  remoteCtrlAddr.resolve( mDestAddr.c_str(), mDestPorts[1], "udp" );

308  // This timeout selected originally in case wanted to kill stream if no Receiver Report

310  Select select( 60 );
311  select.addReader( &rtpCtrlSocket );
312  select.addWriter( &rtpDataSocket );
313 
314  unsigned char buffer[BUFSIZ];
315  while ( !mStop && select.wait() >= 0 )
316  {
317  if ( mStop )
318  break;

320  Select::CommsList readable = select.getReadable();
321  //Info( "rS: %zd, wS: %zd", readable.size(), writeable.size() );
322  if ( writeable.size() > 0 )
323  {
324  mQueueMutex.lock();

326  {
327  for ( FrameQueue::iterator frameIter = mFrameQueue.begin(); frameIter != mFrameQueue.end(); frameIter++ )
328  {
329  packetiseFrame( *frameIter );
330  //delete *frameIter;
331  }
332  mFrameQueue.clear();
333  }
334  mQueueMutex.unlock();
335 
336  mRtpData->lockPacketQueue();

338  if ( !packetQueue.empty() )
339  {
340  bool sendToSocket = false;
341  for ( Select::CommsList::iterator socketIter = writeable.begin(); socketIter != writeable.end(); socketIter++ )
342  if ( UdpInetServer *socket = dynamic_cast<UdpInetServer *>(*socketIter) )
343  if ( socket == &rtpDataSocket )
344  sendToSocket = true;
345  if ( sendToSocket )

348  {

350  /*int nBytes = */rtpDataSocket.write( packet->data(), packet->size() );
351  Debug( 5, "Wrote %zd byte packet to socket", packet->size() );
352  delete *packetIter;
353  }
354  packetQueue.clear();
355  }
356  }
357  mRtpData->unlockPacketQueue();
358  }
359  else if ( readable.size() == 0 )
360  {
361  Error( "RTP timed out" );
362  mStop = true;
363  break;
364  }
365  for ( Select::CommsList::iterator iter = readable.begin(); iter != readable.end(); iter++ )
366  {
367  if ( UdpInetServer *socket = dynamic_cast<UdpInetServer *>(*iter) )
368  {
369  int nBytes = socket->recv( buffer, sizeof(buffer) );
370  Debug( 4, "Read %d bytes on sd %d", nBytes, socket->getReadDesc() );
371  if ( nBytes )
372  {
373  // Process RTP control packets
374  //recvPacket( buffer, nBytes );
375  }
376  else
377  {
378  mStop = true;
379  break;
380  }
381  }
382  else
383  {
384  Panic( "The impossible has happened" );
385  }
386  }
387  // Quite short so we can always keep up with the required packet rate for 25/30 fps
388  usleep( INTERFRAME_TIMEOUT );
389  }
390  rtpCtrlSocket.close();
391  rtpDataSocket.close();
392  }
393  else
394  {
395  while ( !mStop )
396  {
397  if ( mRtspConnection->socket()->isOpen() )
398  {
399  mQueueMutex.lock();
400  if ( !mFrameQueue.empty() )
401  {
402  for ( FrameQueue::iterator frameIter = mFrameQueue.begin(); frameIter != mFrameQueue.end(); frameIter++ )
403  {
404  packetiseFrame( *frameIter );
405  //delete *frameIter;
406  }
407  mFrameQueue.clear();
408 
409  }
410  mQueueMutex.unlock();
411 
412  mRtpData->lockPacketQueue();
413  PacketQueue &packetQueue = mRtpData->packetQueue();
414  for ( PacketQueue::iterator packetIter = packetQueue.begin(); packetIter != packetQueue.end(); packetIter++ )
415  {
416  const ByteBuffer *packet = *packetIter;
417  if ( mRtspConnection->sendInterleavedPacket( mChannels[0], *packet ) )
418  {
419  Debug( 4, "Wrote %zd byte packet on channel %d", packet->size(), mChannels[0] );
420  }
421  else
422  {
423  Error( "Unable to write interleaved packet on channel %d", mChannels[0] );
424  mStop = true;
425  break;
426  }
427  delete *packetIter;
428  }
429  packetQueue.clear();
430  mRtpData->unlockPacketQueue();
431  }
432  // Quite short so we can always keep up with the required packet rate for 25/30 fps
433  usleep( INTERFRAME_TIMEOUT );
434  }
435  }
