44 {
45  ByteBuffer *packet = new ByteBuffer;
46 
47  size_t chunkSize = min(mRtmpConnection->txChunkSize(),payload.size());
48 
49  if ( packetType == 0x00 && (mInitialFrames > 0) )
50  {
51  packetType = 0x40;
52  mInitialFrames--;
53  }
54 
55  switch( packetType )
56  {
57  case 0x00 :
58  {
59  packet->append( chunkStreamStreamId&0x3f );
60 
61  RtmpConnection::ChunkHeader0 header;
62  //header.timestamp = htobe24(mConnection->rtmpTimestamp( timestamp ));
63  //Info( "Timestamp = %u", timestamp );
64  header.timestamp = htobe24( 0 );
65  header.messageLength = htobe24(payload.size());
66  header.messageTypeId = messageTypeId;
67  header.messageStreamId = htole32(mStreamId);
68  packet->append( &header, sizeof(header) );
69 
70  packet->append( payload.data(), chunkSize );
71  payload -= chunkSize;
72  break;
73  }
74  case 0x40 :
75  {
76  packet->append( 0x40|(chunkStreamStreamId&0x3f) );
77 
78  //Info( "Timestamp = %u", timestamp );
79  //uint32_t timestampDelta = timestamp-baseTimestamp;
80  //baseTimestamp = timestamp;
81  //Info( "Timestamp delta = %u", timestampDelta );
82  uint32_t timestampDelta = timestamp;
83  RtmpConnection::ChunkHeader1 header;
84  header.timestampDelta = htobe24(timestampDelta);
85  header.messageLength = htobe24(payload.size());
86  header.messageTypeId = messageTypeId;
87  packet->append( &header, sizeof(header) );
88 
89  packet->append( payload.data(), chunkSize );
90  payload -= chunkSize;
91  break;
92  }
93  case 0xc0 :
94  {
95  packet->append( 0xc0|(chunkStreamStreamId&0x3f) );
96  packet->append( payload.data(), chunkSize );
97  payload -= chunkSize;
98  break;
99  }
100  default :
101  {
102  Fatal( "Unexpected packet type %02x", packetType );
103  }
104  }
105  return( packet );
