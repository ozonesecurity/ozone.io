11 {
12  FILE *cfg;
13  char line[512];
14  char *val;
15  if ( (cfg = fopen( OZ_CONFIG, "r")) == NULL )
16  {
17  Fatal("Can't open %s: %s", OZ_CONFIG, strerror(errno) );
18  }
19  while ( fgets( line, sizeof(line), cfg ) != NULL )
20  {
21  char *line_ptr = line;
22 
23  // Trim off any cr/lf line endings
24  int chomp_len = strcspn( line_ptr, "\r\n" );
25  line_ptr[chomp_len] = '\0';
26 
27  // Remove leading white space
28  int white_len = strspn( line_ptr, " \t" );
29  line_ptr += white_len;
30 
31  // Check for comment or empty line
32  if ( *line_ptr == '\0' || *line_ptr == '#' )
33  continue;
34 
35  // Remove trailing white space
36  char *temp_ptr = line_ptr+strlen(line_ptr)-1;
37  while ( *temp_ptr == ' ' || *temp_ptr == '\t' )
38  {
39  *temp_ptr-- = '\0';
40  temp_ptr--;
41  }
42 
43  // Now look for the '=' in the middle of the line
44  temp_ptr = strchr( line_ptr, '=' );
45  if ( !temp_ptr )
46  {
47  Warning( "Invalid data in %s: '%s'", OZ_CONFIG, line );
48  continue;
49  }
50 
51  // Assign the name and value parts
52  char *name_ptr = line_ptr;
53  char *val_ptr = temp_ptr+1;
54 
55  // Trim trailing space from the name part
56  do
57  {
58  *temp_ptr = '\0';
59  temp_ptr--;
60  }
61  while ( *temp_ptr == ' ' || *temp_ptr == '\t' );
62 
63  // Remove leading white space from the value part
64  white_len = strspn( val_ptr, " \t" );
65  val_ptr += white_len;
66 
67  val = (char *)malloc( strlen(val_ptr)+1 );
68  strncpy( val, val_ptr, strlen(val_ptr)+1 );
69 
70  if ( strcasecmp( name_ptr, "OZ_DB_HOST" ) == 0 )
71  staticConfig.DB_HOST = val;
72  else if ( strcasecmp( name_ptr, "OZ_DB_NAME" ) == 0 )
73  staticConfig.DB_NAME = val;
74  else if ( strcasecmp( name_ptr, "OZ_DB_USER" ) == 0 )
75  staticConfig.DB_USER = val;
76  else if ( strcasecmp( name_ptr, "OZ_DB_PASS" ) == 0 )
77  staticConfig.DB_PASS = val;
78  else if ( strcasecmp( name_ptr, "OZ_PATH_WEB" ) == 0 )
79  staticConfig.PATH_WEB = val;
80  else
81  {
82  // We ignore this now as there may be more parameters than the
83  // c/c++ binaries are bothered about
84  // Warning( "Invalid parameter '%s' in %s", name_ptr, OZ_CONFIG );
85  }
86  }
87  fclose( cfg);
88  config.Assign();
89 }
