107 bool FeedConsumer::waitForProviders()
108 {
109  mProviderMutex.lock();
110  if ( mProviders.empty() )
111  {
112  Debug( 5,"%s: No providers registered for consumer", cidentity() );
113  }
114  mProviderMutex.unlock();
115  int waitCount, readyCount, badCount;
116  do
117  {
118  waitCount = 0;
119  readyCount = 0;
120  badCount = 0;
121  mProviderMutex.lock();
122  Debug( 5,"%s: Got %lu providers", cidentity(), mProviders.size() );
123  for ( ProviderMap::const_iterator iter = mProviders.begin(); iter != mProviders.end(); iter++ )
124  {
125  if ( iter->first->ready() )
126  {
127  readyCount++;
128  }
129  else
130  {
131  if ( iter->first->wait() )
132  {
133  Info( "%s: Provider %s not ready", cidentity(), iter->first->cidentity() );
134  waitCount++;
135  }
136  else
137  {
138  if ( iter->first->ended() )
139  {
140  Error( "%s: Provider %s has ended", cidentity(), iter->first->cidentity() );
141  }
142  else if ( iter->first->error() )
143  {
144  Error( "%s: Provider %s has failed", cidentity(), iter->first->cidentity() );
145  }
146  badCount++;
147  }
148  }
149  }
150  mProviderMutex.unlock();
151  if ( waitCount > 0 && badCount == 0 )
152  {
153  mQueueMutex.lock();
154  mFrameQueue.clear();
155  mQueueMutex.unlock();
156  usleep( 10000 );
157  }
158  } while( waitCount > 0 && badCount == 0 );
159  //Info( "%s: Returning %d", cidentity(), readyCount > 0 && badCount == 0 );
160  return( readyCount > 0 && badCount == 0 );
161 }
