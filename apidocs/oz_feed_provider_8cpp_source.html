170 void FeedProvider::distributeFrame( FramePtr frame )
171 {
172  setReady();
173  mConsumerMutex.lock();
174  ConsumerList badConsumers;
175  for ( ConsumerMap::iterator iter = mConsumers.begin(); iter != mConsumers.end(); iter++ )
176  {
177  if ( iter->second.isQueued() )
178  if ( !iter->second.hasComparators() || iter->second.compare( frame, iter->first ) )
179  if ( !iter->first->queueFrame( frame, this ) )
180  badConsumers.push_back( iter->first );
181  }
182  mConsumerMutex.unlock();
183  for ( ConsumerList::iterator iter = badConsumers.begin(); iter != badConsumers.end(); iter++ )
184  deregisterConsumer( *(*iter) );
185  if ( mCanPoll )
186  {
187  mFrameMutex.lock();
188  mLastFrame = frame;
189  mFrameMutex.unlock();
190  }
191 }
