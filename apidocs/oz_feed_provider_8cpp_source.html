212 void FeedProvider::distributeFrame( FramePtr frame )
213 {
214  setReady();
215  mConsumerMutex.lock();
216  ConsumerList badConsumers;
217  for ( ConsumerMap::iterator iter = mConsumers.begin(); iter != mConsumers.end(); iter++ )
218  {
219  if ( iter->second.isQueued() )
220  if ( !iter->second.hasComparators() || iter->second.compare( frame, iter->first ) )
221  if ( !iter->first->queueFrame( frame, this ) )
222  badConsumers.push_back( iter->first );
223  }
224  mConsumerMutex.unlock();
225  for ( ConsumerList::iterator iter = badConsumers.begin(); iter != badConsumers.end(); iter++ )
226  deregisterConsumer( *(*iter) );
227  if ( mCanPoll )
228  {
229  mFrameMutex.lock();
230  mLastFrame = frame;
231  mFrameMutex.unlock();
232  }
233 }
